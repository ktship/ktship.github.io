01 원클릭 통합 환경 구축 쉘 스크립트

아래 스크립트 전체를 복사해 RunPod 터미널에 붙여넣으면 경로 링크 생성 · 커스텀 노드 설치 · 종속성 해결 · Qwen 2511 모델 다운로드까지 한 번에 완료됩니다.

bash 
#!/bin/bash

# ── 0. 경로 심볼릭 링크 생성 (RunPod 전용)
echo "=== [0/4] ComfyUI 경로 심볼릭 링크 생성 중... ==="
rm -rf /workspace/ComfyUI
ln -s /workspace/runpod-slim/ComfyUI/ /workspace/ComfyUI


# ── 1. 커스텀 노드 설치
echo "=== [1/4] 커스텀 노드 다운로드 중... ==="
cd /workspace/ComfyUI/custom_nodes

git clone https://github.com/jtydhr88/ComfyUI-qwenmultiangle.git
git clone https://github.com/yolain/ComfyUI-Easy-Use.git
git clone https://github.com/chflame163/ComfyUI_LayerStyle.git
git clone https://github.com/hinablue/ComfyUI_3dPoseEditor.git

wget -O "lora_nipple.safetensors" \
  "https://civitai.red/api/download/models/2305397?fileId=2196123&token=여기에_키값"


# ── 2. Python 종속성 설치
echo "=== [2/4] 종속성 Python 라이브러리 설치 중... ==="
pip install -r ComfyUI-Easy-Use/requirements.txt
pip install -r ComfyUI_LayerStyle/requirements.txt
pip install -U "huggingface_hub[cli]"


# ── 3. Hugging Face 모델 다운로드
echo "=== [3/4] Hugging Face에서 모델 다운로드 중... ==="

# 메인 확산 모델 (FP8 Mixed)
hf download Comfy-Org/Qwen-Image-Edit_ComfyUI \
  split_files/diffusion_models/qwen_image_edit_2511_fp8mixed.safetensors \
  --local-dir /workspace/ComfyUI/models/diffusion_models

# 라이트닝 LoRA (4스텝 고속 생성)
hf download lightx2v/Qwen-Image-Edit-2511-Lightning \
  Qwen-Image-Edit-2511-Lightning-4steps-V1.0-fp32.safetensors \
  --local-dir /workspace/ComfyUI/models/loras

# 텍스트 인코더 (Qwen 2.5 VL 7B FP8)
hf download Comfy-Org/Qwen-Image_ComfyUI \
  split_files/text_encoders/qwen_2.5_vl_7b_fp8_scaled.safetensors \
  --local-dir /workspace/ComfyUI/models/text_encoders

# VAE
hf download Comfy-Org/Qwen-Image_ComfyUI \
  split_files/vae/qwen_image_vae.safetensors \
  --local-dir /workspace/ComfyUI/models/vae

# 다각도 뷰 LoRA
hf download fal/Qwen-Image-Edit-2511-Multiple-Angles-LoRA \
  qwen-image-edit-2511-multiple-angles-lora.safetensors \
  --local-dir /workspace/ComfyUI/models/loras


# ── 4. split_files 경로 정리
echo "=== [4/4] 모델 위치 정리 중... ==="
mv /workspace/ComfyUI/models/diffusion_models/split_files/diffusion_models/* \
   /workspace/ComfyUI/models/diffusion_models/ 2>/dev/null
mv /workspace/ComfyUI/models/text_encoders/split_files/text_encoders/* \
   /workspace/ComfyUI/models/text_encoders/ 2>/dev/null
mv /workspace/ComfyUI/models/vae/split_files/vae/* \
   /workspace/ComfyUI/models/vae/ 2>/dev/null

echo "=== 완료! ComfyUI 서버를 재시작해 주세요. ==="
